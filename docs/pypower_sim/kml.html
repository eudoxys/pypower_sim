pypower_sim.kml

KML generator

The KML class collect KML entities and saves them to a file when the close () method is called. Note that the KML file is not saved if the KML object is deleted without being closed.

The following methods are used to create KML files:

  • pypower_sim.kml.KML.add_linestyle creates a linestyle for use by line entities

  • pypower_sim.kml.KML.add_markerstyle creates a markerstyle for use by market entities

  • pypower_sim.kml.KML.add_line creates a line entity

  • pypower_sim.kml.KML.add_marker creates a marker entity

See also

  • Google Earth KML Documentation

134    def close(self,write=True):
135        """Close KML file
136
137        # Arguments
138
139        - `write`: enable output to `pypower_sim.kml.KML.kmlfile`
140        """
141        if self.kmlfile and write:
142            with open(self.kmlfile,"w",encoding="utf-8") as fh:
143
144                print('<?xml version="1.0" encoding="UTF-8"?>',file=fh)
145                print('<kml xmlns="http://www.opengis.net/kml/2.2"'
146                    ' xmlns:gx="http://www.google.com/kml/ext/2.2">',file=fh)
147                print("<Document>",file=fh)
148                print(f"  <name>{self.name}</name>",file=fh)
149
150                # output marker styles
151                for name,data in self.markerstyle.items():
152                    print(f'  <Style id="{name}">',file=fh)
153                    print("    <IconStyle>",file=fh)
154                    if "scale" in data:
155                        print(f"      <scale>{data['scale']}</scale>",file=fh)
156                    print("      <Icon>",file=fh)
157                    print(f"        <href>{data['url']}</href>",file=fh)
158                    print("      </Icon>",file=fh)
159                    print("    </IconStyle>",file=fh)
160                    print("  </Style>",file=fh)
161
162
163                # output line styles
164                for name,data in self.linestyle.items():
165                    print(f'  <Style id="{name}">',file=fh)
166                    print("    <PolyStyle>",file=fh)
167                    print(f"      <color>{data['color']}</color>",file=fh)
168                    print("    </PolyStyle>",file=fh)
169                    print("    <LineStyle>",file=fh)
170                    print(f"      <color>{data['color']}</color>",file=fh)
171                    print(f"      <width>{data['width']}</width>",file=fh)
172                    print("    </LineStyle>",file=fh)
173                    print("  </Style>",file=fh)
174
175                # output markers
176                for name,data in self.marker.items():
177                    print("  <Placemark>",file=fh)
178                    if name:
179                        print(f"    <name>{name}</name>""",file=fh)
180                    if "style" in data:
181                        print(f"    <styleUrl>#{data['style']}</styleUrl>",file=fh)
182                    print(f"""    <Point><coordinates>{','.join(f'{x}'
183                        for x in data['position'])}</coordinates></Point>""",file=fh)
184                    print("  </Placemark>",file=fh)
185
186                # output lines
187                for name,data in self.line.items():
188                    print("  <Placemark>",file=fh)
189                    if name:
190                        print(f"    <name>{name}</name>",file=fh)
191                    if "style" in data:
192                        print(f"    <styleUrl>#{data['style']}</styleUrl>",file=fh)
193                    print("    <LineString>",file=fh)
194                    print("      <tesselate>1</tesselate>",file=fh)
195                    print("      <coordinates>",file=fh)
196                    print(f"""      {','.join(f'{x}'
197                        for x in data['from_position'])}""",file=fh)
198                    print(f"""      {','.join(f'{x}'
199                        for x in data['to_position'])}""",file=fh)
200                    print("      </coordinates>",file=fh)
201                    print("    </LineString>",file=fh)
202                    print("  </Placemark>",file=fh)
203                print("</Document>",file=fh)
204                print("</kml>""",file=fh)
205
206        self.kmlfile = None
